create table member (
  m_id varchar(10) unsigned not null,
  m_password varchar(10) not null,
  m_name varchar(10) not null,
  m_tel varchar(20) not null,
  m_field varchar(16) not null,
  hashed varchar(256), 
  primary key(m_id)
) default charset=utf8;
desc member;

create table product (
  p_id int unsigned not null auto_increment,
  p_name varchar(10) not null,
  p_img varchar(80) not null,
  p_price varchar(16) not null,
  p_quantity int not null,
  primary key(p_id)
) auto_increment=1 default charset=utf8;
desc product;

create table order (
  o_id int unsigned not null auto_increment,
  o_name varchar(10) not null,
  o_tel varchar(20) not null,
  o_address varchar(80) not null,
  p_id int not null,
  p_name varchar(10) not null,
  o_quantity int not null, 
  primary key(o_id)
) auto_increment=10001 default charset=utf8;
desc order;
